int main () {
     int n1;
     int i1;
     int k1;
     int a1 [(-6 + 10 + 2)] = {100, (38 + 10 + 2), 20, 10, (-7 + 10 + 2), (-11 + 10 + 2)};
     int b1 [(-6 + 10 + 2)];
     if (0) {
         int x, y;
         for (x = 0; x < 20; x++) {
             y += x;
         }

             return y;
     };

         k1 = 0;
